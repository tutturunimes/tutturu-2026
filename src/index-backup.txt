export async function getStaticProps() {
  try {
    const path = process.env.NEXT_PUBLIC_ABSOLUTE_PATH;
   
    const response = await axios.get(`${path}/api/v1/ongoing?page=1`, {
      next: {
        revalidate: 10, 
        cache: 'force-cache',
      },
    });

    const completed = await axios.get(`${path}/api/v1/complete?page=1`, {
      next: {
        revalidate: 10, 
        cache: 'force-cache',
      },
    });
    
    const jadwal = await axios.get(`${path}/api/v1/jadwal`, {
      next: {
        revalidate: 10, 
        cache: 'force-cache',
      },
    });
    
    // GET BANNER
    const aniDetail = await axios.get(`${path}/api/v1/detail?title=${response?.data?.ongoing[0]?.slug}`, {
      next: {
        revalidate: 10, 
        cache: 'force-cache',
      },
    });
    const originalTitle = aniDetail.data.info.filter(item => item.startsWith("Judul:"));
    const title = originalTitle.join("").split(":")[1].trim();
    const banner = await getAnilistByTitle(title);
    // END GET BANNER

    if (response.status === 200) {
 
      return {
        props: {
          data: {
            ongoing: response.data.ongoing,
            completed: completed.data.batch,
            jadwal: jadwal.data,
            banner: banner.data.data.Media,
          },
        },
        revalidate: 60 * 60, // Revalidate every hour (adjust this as needed)
      };
    } else {
      throw new Error('Failed to fetch data');
    }
  } catch (error) {
    console.error('Failed to fetch or parse HTML:', error);
    return {
      props: {
        error: 'Failed to fetch data',
      },
    };
  }
}
